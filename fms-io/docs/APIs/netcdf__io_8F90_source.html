256 !> @brief Open a netcdf file.
257 subroutine netcdf_file_open(fileobj, &
258  path, &
259  mode, &
260  nc_format, &
261  pelist, &
262  is_restart)

264  character(len=*),intent(in) :: path !< File path.
265  character(len=*),intent(in) :: mode !< File mode. Allowed values are:
266  !! "read", "append", "write", or
267  !! "overwrite".
268  character(len=*),intent(in),optional :: nc_format !< Netcdf format that
269  !! new files are written
270  !! as. Allowed values
271  !! are: "64bit", "classic",
272  !! or "netcdf4". Defaults to
273  !! "64bit".
274  integer,dimension(:),intent(in),optional :: pelist !< List of ranks associated
275  !! with this file. If not
276  !! provided, only the current
277  !! rank will be able to
278  !! act on the file.
279  logical,intent(in),optional :: is_restart !< Flag telling if this file
280  !! is a restart file. Defaults
281  !! to false.
282  integer :: nc_format_param
283  integer :: err
284  call string_copy(fileobj%path, &
285  trim(path))
286  if (allocated(fileobj%pelist)) then
287  deallocate(fileobj%pelist)
288  endif
289  if (present(pelist)) then
290  allocate(fileobj%pelist(size(pelist)))
291  fileobj%pelist = pelist
292  else
293  allocate(fileobj%pelist(1))
294  fileobj%pelist(1) = mpp_pe()
295  endif
296  fileobj%io_root = fileobj%pelist(1)
297  if (mpp_pe() .eq. fileobj%io_root) then
298  fileobj%is_root = .true.
299  else
300  fileobj%is_root = .false.
301  endif
302  if (present(is_restart)) then

304  else
305  fileobj%is_restart = .false.
306  endif
307  if (fileobj%is_restart) then
308  call restart_filepath_mangle(fileobj%path, &
309  fileobj%path)
310  allocate(fileobj%restart_vars(max_num_restart_vars))
311  fileobj%num_restart_vars = 0
312  endif
313  if (fileobj%is_root) then
314  if (present(nc_format)) then
315  if (string_compare(nc_format,"64bit",.true.)) then
316  nc_format_param = nf90_64bit_offset
317  elseif (string_compare(nc_format,"classic",.true.)) then
318  nc_format_param = nf90_classic_model
319  elseif (string_compare(nc_format,"netcdf4",.true.)) then
320  nc_format_param = nf90_hdf5
321  else
322  call error("unrecognized netcdf file format " &
323  //trim(nc_format)//".")
324  endif
325  else
326  nc_format_param = nf90_64bit_offset
327  endif
328  if (string_compare(mode,"read",.true.)) then
329  err = nf90_open(trim(fileobj%path), &
330  nf90_nowrite, &
331  fileobj%ncid)
332  elseif (string_compare(mode,"append",.true.)) then
333  err = nf90_open(trim(fileobj%path), &
334  nf90_write, &
335  fileobj%ncid)
336  elseif (string_compare(mode,"write",.true.)) then
337  err = nf90_create(trim(fileobj%path), &
338  ior(nf90_noclobber,nc_format_param), &
339  fileobj%ncid)
340  elseif (string_compare(mode,"overwrite",.true.)) then
341  err = nf90_create(trim(fileobj%path), &
342  ior(nf90_clobber,nc_format_param), &
343  fileobj%ncid)
344  else
345  call error("unrecognized file mode "//trim(mode)//".")
346  endif
347  call check_netcdf_code(err)
348  else
349  fileobj%ncid = missing_ncid
350  endif
351 end subroutine netcdf_file_open
